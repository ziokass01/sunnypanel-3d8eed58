static std::string g_sunny_ca_path;

static inline void Sunny_EnsureEmbeddedCaBundleInstalled() {
    if (!g_sunny_ca_path.empty()) return;

    const std::string dir = GetFilesDirPathSafe();
    if (dir.empty()) return;

    g_sunny_ca_path = dir + "/sunny_cacert.pem";

    // Nếu đã có file CA thì dùng luôn
    if (access(g_sunny_ca_path.c_str(), R_OK) == 0) return;

    std::ofstream out(g_sunny_ca_path, std::ios::binary);
    if (!out.good()) {
        g_sunny_ca_path.clear();
        return;
    }

    out.write(SUNNY_CA_BUNDLE_PEM, (std::streamsize)(sizeof(SUNNY_CA_BUNDLE_PEM) - 1));
    out.close();

    __android_log_print(ANDROID_LOG_INFO, "SunnySSL", "Installed CA bundle: %s", g_sunny_ca_path.c_str());
}

static inline const char* Sunny_GetCaBundlePath() {
    Sunny_EnsureEmbeddedCaBundleInstalled();
    return g_sunny_ca_path.empty() ? "" : g_sunny_ca_path.c_str();
}

long getAvailableRAM()
{
    std::ifstream meminfo("/proc/meminfo");
    std::string line;
    long ram = 0;
    while (std::getline(meminfo, line)) {
        if (line.find("MemAvailable:") == 0) {
            sscanf(line.c_str(), "MemAvailable: %ld kB", &ram);
            break;
        }
    }
    return ram / 1024; // MB
}

float getAvailableStorage(const char* path = "/data")
{
    struct statvfs stat;
    if (statvfs(path, &stat) != 0)
        return 0.0f;
    return (float)(stat.f_bavail * stat.f_frsize) / (1024*1024*1024); // GB
}

static inline long long GetRemainingSecondsNow() {
    if (g_remaining_seconds_at_sync < 0) return g_remaining_seconds_at_sync;
    const auto elapsed = std::chrono::duration_cast<std::chrono::seconds>(
        std::chrono::steady_clock::now() - g_remaining_sync_tp).count();
    long long rem = g_remaining_seconds_at_sync - (long long)elapsed;
    if (rem < 0) rem = 0;
    return rem;
}

static inline std::string FormatRemainingDHMS(long long seconds) {
    if (seconds < 0) return "—";
    long long s = seconds;
    long long d = s / 86400; s %= 86400;
    long long h = s / 3600;  s %= 3600;
    long long m = s / 60;    s %= 60;

    char buf[64];
    if (d > 0) snprintf(buf, sizeof(buf), "%lldd %lldh %lldm", d, h, m);
    else if (h > 0) snprintf(buf, sizeof(buf), "%lldh %lldm", h, m);
    else snprintf(buf, sizeof(buf), "%lldm %llds", m, s);
    return std::string(buf);
}

static inline std::string Sunny_Trim(const std::string& s) {
    size_t b = 0, e = s.size();
    while (b < e && (s[b] == ' ' || s[b] == '\n' || s[b] == '\r' || s[b] == '\t')) b++;
    while (e > b && (s[e - 1] == ' ' || s[e - 1] == '\n' || s[e - 1] == '\r' || s[e - 1] == '\t')) e--;
    return s.substr(b, e - b);
}

static inline bool Sunny_JsonGetToken(const std::string& json, const char* key, std::string& outTok) {
    if (!key) return false;
    const std::string pat = std::string("\"") + key + "\"";
    size_t p = json.find(pat);
    if (p == std::string::npos) return false;
    p = json.find(':', p + pat.size());
    if (p == std::string::npos) return false;
    p++;
    while (p < json.size() && (json[p] == ' ' || json[p] == '\n' || json[p] == '\r' || json[p] == '\t')) p++;
    if (p >= json.size()) return false;

    // String token
    if (json[p] == '"') {
        std::string out;
        p++; // after opening quote
        while (p < json.size()) {
            char c = json[p++];
            if (c == '\\' && p < json.size()) {
                char esc = json[p++];
                switch (esc) {
                    case '"': out.push_back('"'); break;
                    case '\\': out.push_back('\\'); break;
                    case '/': out.push_back('/'); break;
                    case 'b': out.push_back('\b'); break;
                    case 'f': out.push_back('\f'); break;
                    case 'n': out.push_back('\n'); break;
                    case 'r': out.push_back('\r'); break;
                    case 't': out.push_back('\t'); break;
                    default: out.push_back(esc); break;
                }
                continue;
            }
            if (c == '"') break;
            out.push_back(c);
        }
        outTok = out;
        return true;
    }

    // Non-string token: read until ',' or '}'
    size_t end = json.find_first_of(",}", p);
    if (end == std::string::npos) end = json.size();
    outTok = Sunny_Trim(json.substr(p, end - p));
    return true;
}

static inline bool Sunny_JsonGetBool(const std::string& json, const char* key, bool& outVal) {
    std::string tok;
    if (!Sunny_JsonGetToken(json, key, tok)) return false;
    if (tok == "true") { outVal = true; return true; }
    if (tok == "false") { outVal = false; return true; }
    return false;
}

static inline bool Sunny_JsonGetInt64(const std::string& json, const char* key, long long& outVal) {
    std::string tok;
    if (!Sunny_JsonGetToken(json, key, tok)) return false;
    if (tok.empty() || tok == "null") return false;
    char* endp = nullptr;
    long long v = strtoll(tok.c_str(), &endp, 10);
    if (endp == tok.c_str()) return false;
    outVal = v;
    return true;
}

static inline bool Sunny_JsonGetString(const std::string& json, const char* key, std::string& outVal) {
    std::string tok;
    if (!Sunny_JsonGetToken(json, key, tok)) return false;
    if (tok == "null") { outVal.clear(); return true; }
    outVal = tok;
    return true;
}

static inline bool ParseVerifyResponseToState(const std::string& response, std::string* errOut) {
    if (errOut) errOut->clear();

    LicState st;
    st.ok = false;
    st.msg.clear();
    st.expires_at.clear();
    st.server_time.clear();
    st.max_devices = 0;
    st.started = false;
    st.has_remaining = false;
    st.remaining_seconds = 0;

    bool okField = false;
    if (!Sunny_JsonGetBool(response, "ok", okField)) {
        if (errOut) *errOut = "Bad response: missing ok";
        gLicState = st;
        return false;
    }
    st.ok = okField;

    (void)Sunny_JsonGetString(response, "msg", st.msg);
    (void)Sunny_JsonGetString(response, "expires_at", st.expires_at);
    (void)Sunny_JsonGetString(response, "server_time", st.server_time);

    long long maxDev = 0;
    if (Sunny_JsonGetInt64(response, "max_devices", maxDev)) {
        if (maxDev < 0) maxDev = 0;
        st.max_devices = (int)maxDev;
    }

    bool started = false;
    if (Sunny_JsonGetBool(response, "started", started)) st.started = started;

    std::string remTok;
    if (Sunny_JsonGetToken(response, "remaining_seconds", remTok) && !remTok.empty() && remTok != "null") {
        char* endp = nullptr;
        long long rem = strtoll(remTok.c_str(), &endp, 10);
        if (endp != remTok.c_str()) {
            st.has_remaining = true;
            st.remaining_seconds = rem;
        }
    }

    gLicState = st;

    g_remaining_sync_tp = std::chrono::steady_clock::now();
    if (st.has_remaining) {
        g_remaining_seconds_at_sync = st.remaining_seconds;
    } else {
        g_remaining_seconds_at_sync = -1; 
    }

    if (!st.ok) {
        if (errOut) *errOut = st.msg.empty() ? "License invalid" : st.msg;
        return false;
    }
    return true;
}

static inline bool LicenseIsExpiredOrInvalid(std::string* out_reason = nullptr) {
    const long long rem = GetRemainingSecondsNow();
    if (rem == 0) {
        if (out_reason) *out_reason = "❌ Key đã hết hạn. Vui lòng gia hạn.";
        return true;
    }
    if (rem < 0) {
#if SUNNY_ALLOW_LIFETIME_KEYS
        return false;
#else
        if (out_reason) *out_reason = "❌ Key không có thời hạn (expires_at = null). Vui lòng tạo key có thời hạn.";
        return true;
#endif
    }
    return false;
}

static inline bool Sunny_IsValidKeyFormatSafe(const std::string& k) {
    if (k.size() != 20) return false; 
    auto isAZ09 = [](char c) {
        return (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9');
    };
    if (k.rfind("SUNNY-", 0) != 0) return false;
    for (int i = 5; i < 20; i++) {
        if (i == 5 || i == 10 || i == 15) {
            if (k[i] != '-') return false;
        } else {
            if (!isAZ09(k[i])) return false;
        }
    }
    return true;
}

static inline bool JsonGetBool(const std::string& s, const char* key, bool& out) {
    std::string k = std::string("\"") + key + "\"";
    size_t p = s.find(k);
    if (p == std::string::npos) return false;
    p = s.find(':', p);
    if (p == std::string::npos) return false;
    p++;
    while (p < s.size() && (s[p] == ' ' || s[p] == '\t' || s[p] == '\n' || s[p] == '\r')) p++;
    if (s.compare(p, 4, "true") == 0)  { out = true;  return true; }
    if (s.compare(p, 5, "false") == 0) { out = false; return true; }
    return false;
}

static inline bool JsonGetInt64(const std::string& s, const char* key, long long& out) {
    std::string k = std::string("\"") + key + "\"";
    size_t p = s.find(k);
    if (p == std::string::npos) return false;
    p = s.find(':', p);
    if (p == std::string::npos) return false;
    p++;
    while (p < s.size() && (s[p] == ' ' || s[p] == '\t' || s[p] == '\n' || s[p] == '\r')) p++;
    if (s.compare(p, 4, "null") == 0) return false;
    bool neg = false;
    if (p < s.size() && s[p] == '-') { neg = true; p++; }
    long long val = 0;
    bool any = false;
    while (p < s.size() && s[p] >= '0' && s[p] <= '9') {
        any = true;
        val = val * 10 + (s[p] - '0');
        p++;
    }
    if (!any) return false;
    out = neg ? -val : val;
    return true;
}

static inline bool JsonGetString(const std::string& s, const char* key, std::string& out) {
    std::string k = std::string("\"") + key + "\"";
    size_t p = s.find(k);
    if (p == std::string::npos) return false;
    p = s.find(':', p);
    if (p == std::string::npos) return false;
    p++;
    while (p < s.size() && (s[p] == ' ' || s[p] == '\t' || s[p] == '\n' || s[p] == '\r')) p++;
    if (s.compare(p, 4, "null") == 0) return false;

    return true;
}

static inline std::string Sunny_GetAndroidIdJNI() {
    if (!jvm1) return "";

    JNIEnv* env = nullptr;
    bool needsDetach = false;
    jint jniStat = jvm1->GetEnv((void**)&env, JNI_VERSION_1_6); 
    if (jniStat == JNI_EDETACHED) {
        if (jvm1->AttachCurrentThread(&env, nullptr) != JNI_OK) return "";
        needsDetach = true;
    }
    if (!env) return "";

    std::string out;

    jobject ctx = GetContextSafe(env);
    if (!ctx) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jclass ctxCls = env->GetObjectClass(ctx);
    jmethodID midGetCR = ctxCls ? env->GetMethodID(ctxCls, "getContentResolver", "()Landroid/content/ContentResolver;") : nullptr;
    JniClearException(env);

    jobject cr = (midGetCR ? env->CallObjectMethod(ctx, midGetCR) : nullptr);
    JniClearException(env);

    if (ctxCls) env->DeleteLocalRef(ctxCls);

    if (!cr) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jclass secureCls = env->FindClass("android/provider/Settings$Secure");
    JniClearException(env);

    if (!secureCls) {
        env->DeleteLocalRef(cr);
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jmethodID midGetString = env->GetStaticMethodID(
        secureCls,
        "getString",
        "(Landroid/content/ContentResolver;Ljava/lang/String;)Ljava/lang/String;"
    );
    JniClearException(env);

    jstring androidIdKey = env->NewStringUTF("android_id");
    jstring jId = (midGetString ? (jstring)env->CallStaticObjectMethod(secureCls, midGetString, cr, androidIdKey) : nullptr);
    JniClearException(env);

    if (androidIdKey) env->DeleteLocalRef(androidIdKey);
    env->DeleteLocalRef(cr);
    env->DeleteLocalRef(secureCls);

    if (jId) {
        const char* c = env->GetStringUTFChars(jId, nullptr);
        if (c) {
            out.assign(c);
            env->ReleaseStringUTFChars(jId, c);
        }
        env->DeleteLocalRef(jId);
    }

    auto trim = [](std::string& s) {
        size_t start = s.find_first_not_of(" \t\r\n");
        size_t end = s.find_last_not_of(" \t\r\n");
        if (start == std::string::npos) { s.clear(); return; }
        s = s.substr(start, end - start + 1);
    };
    trim(out);

    if (needsDetach) jvm1->DetachCurrentThread();
    return out;
}

std::string getDeviceID() {
    std::string id = Sunny_GetAndroidIdJNI();

    if (id.empty()) {
        FILE* fp = popen("settings get secure android_id", "r");
        if (fp) {
            char buf[128] = {0};
            fgets(buf, sizeof(buf), fp);
            pclose(fp);

            id = buf;
            id.erase(std::remove(id.begin(), id.end(), '\n'), id.end());
            id.erase(std::remove(id.begin(), id.end(), '\r'), id.end());
        }
    }

    if (id.empty()) id = "unknown_device";

    if (id.size() > 120) id.resize(120);
    return id;
}

        0x19a4c116ul,0x1e376c08ul,0x2748774cul,0x34b0bcb5ul,0x391c0cb3ul,0x4ed8aa4aul,0x5b9cca4ful,0x682e6ff3ul,
        0x748f82eeul,0x78a5636ful,0x84c87814ul,0x8cc70208ul,0x90befffaul,0xa4506cebul,0xbef9a3f7ul,0xc67178f2ul
    };

    uint32_t h[8] = {
        0x6a09e667ul,0xbb67ae85ul,0x3c6ef372ul,0xa54ff53aul,
        0x510e527ful,0x9b05688cul,0x1f83d9abul,0x5be0cd19ul
    };

    auto read_be32 = [](const uint8_t* p)->uint32_t {
        return ((uint32_t)p[0] << 24) | ((uint32_t)p[1] << 16) | ((uint32_t)p[2] << 8) | (uint32_t)p[3];
    };
    auto write_be32 = [](uint8_t* p, uint32_t v) {
        p[0] = (uint8_t)(v >> 24); p[1] = (uint8_t)(v >> 16); p[2] = (uint8_t)(v >> 8); p[3] = (uint8_t)v;
    };

    uint8_t block[64];
    size_t off = 0;
    uint64_t bit_len = (uint64_t)len * 8ull;

    auto compress = [&](const uint8_t* b) {
        uint32_t w[64];
        for (int i = 0; i < 16; ++i) w[i] = read_be32(b + i*4);
        for (int i = 16; i < 64; ++i) {
            uint32_t s0 = Sunny_ROR32(w[i-15],7) ^ Sunny_ROR32(w[i-15],18) ^ (w[i-15] >> 3);
            uint32_t s1 = Sunny_ROR32(w[i-2],17) ^ Sunny_ROR32(w[i-2],19) ^ (w[i-2] >> 10);
            w[i] = w[i-16] + s0 + w[i-7] + s1;
        }
        uint32_t a=h[0],b2=h[1],c=h[2],d=h[3],e=h[4],f=h[5],g=h[6],hh=h[7];
        for (int i = 0; i < 64; ++i) {
            uint32_t S1 = Sunny_ROR32(e,6) ^ Sunny_ROR32(e,11) ^ Sunny_ROR32(e,25);
            uint32_t ch = (e & f) ^ ((~e) & g);
            uint32_t temp1 = hh + S1 + ch + k[i] + w[i];
            uint32_t S0 = Sunny_ROR32(a,2) ^ Sunny_ROR32(a,13) ^ Sunny_ROR32(a,22);
            uint32_t maj = (a & b2) ^ (a & c) ^ (b2 & c);
            uint32_t temp2 = S0 + maj;
            hh=g; g=f; f=e; e=d + temp1; d=c; c=b2; b2=a; a=temp1 + temp2;
        }
        h[0]+=a; h[1]+=b2; h[2]+=c; h[3]+=d; h[4]+=e; h[5]+=f; h[6]+=g; h[7]+=hh;
    };

    while (off + 64 <= len) {
        compress(data + off);
        off += 64;
    }

    size_t rem = len - off;
    for (size_t i = 0; i < rem; ++i) block[i] = data[off + i];
    block[rem++] = 0x80;
    if (rem > 56) {
        for (size_t i = rem; i < 64; ++i) block[i] = 0;
        compress(block);
        rem = 0;
    }
    for (size_t i = rem; i < 56; ++i) block[i] = 0;
    for (int i = 0; i < 8; ++i) block[63 - i] = (uint8_t)(bit_len >> (8*i));
    compress(block);

    for (int i = 0; i < 8; ++i) write_be32(out + i*4, h[i]);
}

static inline std::string Sunny_SHA256Hex(const std::string& s) {
    uint8_t out[32];
    Sunny_SHA256((const uint8_t*)s.data(), s.size(), out);
    return Sunny_ToHex(out, 32);
}

static inline std::string Sunny_HmacSha256Hex(const std::string& key, const std::string& msg) {
    const size_t BLOCK = 64;
    uint8_t k0[BLOCK];
    memset(k0, 0, sizeof(k0));

    if (key.size() > BLOCK) {
        uint8_t kh[32];
        Sunny_SHA256((const uint8_t*)key.data(), key.size(), kh);
        memcpy(k0, kh, 32);
    } else {
        memcpy(k0, key.data(), key.size());
    }

    uint8_t ipad[BLOCK], opad[BLOCK];
    for (size_t i = 0; i < BLOCK; ++i) {
        ipad[i] = k0[i] ^ 0x36;
        opad[i] = k0[i] ^ 0x5c;
    }

    std::string inner_input;
    inner_input.resize(BLOCK + msg.size());
    memcpy(&inner_input[0], ipad, BLOCK);
    memcpy(&inner_input[BLOCK], msg.data(), msg.size());

    uint8_t inner_hash[32];
    Sunny_SHA256((const uint8_t*)inner_input.data(), inner_input.size(), inner_hash);

    uint8_t outer_input[BLOCK + 32];
    memcpy(outer_input, opad, BLOCK);
    memcpy(outer_input + BLOCK, inner_hash, 32);

    uint8_t out[32];
    Sunny_SHA256(outer_input, sizeof(outer_input), out);
    return Sunny_ToHex(out, 32);
}

static inline std::string Sunny_GenNonceHex(size_t bytes = 16) {
    std::string buf(bytes, '\0');
    int fd = open("/dev/urandom", O_RDONLY);
    if (fd >= 0) {
        ssize_t r = read(fd, &buf[0], bytes);
        close(fd);
        if (r != (ssize_t)bytes) {
            for (size_t i = 0; i < bytes; ++i) buf[i] = (char)(rand() & 0xFF);
        }
    } else {
        srand((unsigned)time(nullptr) ^ (unsigned)getpid());
        for (size_t i = 0; i < bytes; ++i) buf[i] = (char)(rand() & 0xFF);
    }
    return Sunny_ToHex((const uint8_t*)buf.data(), buf.size());
}

static inline std::string Sunny_JsonEscape(const std::string& in) {
    std::ostringstream oss;
    for (unsigned char c : in) {
        switch (c) {
            case '\\': oss << "\\\\"; break;
            case '"':  oss << "\\\""; break;
            case '\b': oss << "\\b"; break;
            case '\f': oss << "\\f"; break;
            case '\n': oss << "\\n"; break;
            case '\r': oss << "\\r"; break;
            case '\t': oss << "\\t"; break;
            default:
                if (c < 0x20) {
                    oss << "\\u" << std::hex << std::setw(4) << std::setfill('0') << (int)c;
                } else {
                    oss << (char)c;
                }
        }
    }
    return oss.str();
}

static inline std::string GetDeviceNameSafe() {
    char model[128] = {0};
    FILE* fp = popen("getprop ro.product.model", "r");
    if (fp) {
        fgets(model, sizeof(model)-1, fp);
        pclose(fp);
    }
    std::string s(model);
    while (!s.empty() && (s.back() == '\n' || s.back() == '\r')) s.pop_back();
    if (s.empty()) s = "ANDROID";
    return s;
}

static inline bool Sunny_PathReadable_(const char* p, int mode) {
    return (p && access(p, mode) == 0);
}

static inline std::string Sunny_FindSystemCaDir_() {
    const char* candidates[] = {
        "/apex/com.android.conscrypt/cacerts",
        "/system/etc/security/cacerts",
    };
    for (const char* c : candidates) {
        if (Sunny_PathReadable_(c, R_OK | X_OK)) return std::string(c);
    }
    return std::string();
}

static inline std::string Sunny_FindCaBundleFile_() {
    Sunny_EnsureEmbeddedCaBundleInstalled();
    const char* embedded = Sunny_GetCaBundlePath();
    if (embedded && embedded[0] && Sunny_PathReadable_(embedded, R_OK)) return std::string(embedded);

    const char* candidates[] = {
        "/sdcard/cacert.pem",
        "/sdcard/Download/cacert.pem",
        "/data/local/tmp/cacert.pem",
    };
    for (const char* c : candidates) {
        if (Sunny_PathReadable_(c, R_OK)) return std::string(c);
    }
    return std::string();
}

static inline void Sunny_ApplyCaToCurl_(CURL* curl) {
    if (!curl) return;
    const std::string cafile = Sunny_FindCaBundleFile_();
    if (!cafile.empty()) {
        curl_easy_setopt(curl, CURLOPT_CAINFO, cafile.c_str());
        return;
    }

    const std::string capath = Sunny_FindSystemCaDir_();
    if (!capath.empty()) {
        curl_easy_setopt(curl, CURLOPT_CAPATH, capath.c_str());
    }

#if defined(CURLOPT_SSL_OPTIONS) && defined(CURLSSLOPT_NATIVE_CA)
    curl_easy_setopt(curl, CURLOPT_SSL_OPTIONS, (long)CURLSSLOPT_NATIVE_CA);
#endif
}

std::string send_request(const std::string& key,
                        const std::string& device,
                        const std::string& device_name,
                        std::string* out_err = nullptr) {
    if (out_err) out_err->clear();

    static std::once_flag g_curl_once;
    std::call_once(g_curl_once, []() {
        curl_global_init(CURL_GLOBAL_DEFAULT);
    });

    CURL* curl = curl_easy_init();
    if (!curl) {
        if (out_err) *out_err = "CURL_INIT_FAILED";
        return "";
    }
    Sunny_EnsureEmbeddedCaBundleInstalled();

    char errbuf[CURL_ERROR_SIZE];
    errbuf[0] = 0;
    curl_easy_setopt(curl, CURLOPT_ERRORBUFFER, errbuf);

    std::string response;

    std::string body = "{\"key\":\"" + Sunny_JsonEscape(key) +
                       "\",\"device\":\"" + Sunny_JsonEscape(device) +
                       "\",\"device_name\":\"" + Sunny_JsonEscape(device_name) + "\"}";

    std::string ts = std::to_string((long long)time(nullptr));     
    std::string nonce = Sunny_GenNonceHex(16);                     
    std::string body_hash = Sunny_SHA256Hex(body);
    std::string canonical = ts + "." + nonce + "." + body_hash;

    std::string secret = GetSunnySecret();
    
    std::string sig = Sunny_HmacSha256Hex(secret, canonical);

    struct curl_slist* headers = nullptr;
    headers = curl_slist_append(headers, "Content-Type: application/json");
    headers = curl_slist_append(headers, "Accept: application/json");

    std::string h1 = "x-ts: " + ts;
    std::string h2 = "x-nonce: " + nonce;
    std::string h3 = "x-sig: " + sig;

    headers = curl_slist_append(headers, h1.c_str());
    headers = curl_slist_append(headers, h2.c_str());
    headers = curl_slist_append(headers, h3.c_str());

    curl_easy_setopt(curl, CURLOPT_URL, API_URL.c_str());
    curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
    curl_easy_setopt(curl, CURLOPT_POSTFIELDS, body.c_str());
    curl_easy_setopt(curl, CURLOPT_POSTFIELDSIZE, (long)body.size());
    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteCallback);
    curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response);
    curl_easy_setopt(curl, CURLOPT_USERAGENT, "SunnyMod/1.0");
    curl_easy_setopt(curl, CURLOPT_ACCEPT_ENCODING, ""); 
    curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, 1L);
    curl_easy_setopt(curl, CURLOPT_IPRESOLVE, CURL_IPRESOLVE_V4);
    curl_easy_setopt(curl, CURLOPT_HTTP_VERSION, CURL_HTTP_VERSION_1_1);

    curl_easy_setopt(curl, CURLOPT_CONNECTTIMEOUT, 10L);
    curl_easy_setopt(curl, CURLOPT_TIMEOUT, 15L);
    curl_easy_setopt(curl, CURLOPT_NOSIGNAL, 1L);

    auto setup_ssl = [&](bool insecure) {
    curl_easy_setopt(curl, CURLOPT_SSL_VERIFYPEER, insecure ? 0L : 1L);
    curl_easy_setopt(curl, CURLOPT_SSL_VERIFYHOST, insecure ? 0L : 2L);

    if (!insecure) {
        const char* ca = Sunny_GetCaBundlePath();
        if (ca && ca[0]) {
            curl_easy_setopt(curl, CURLOPT_CAINFO, ca);
        }

        Sunny_ApplyCaToCurl_(curl);
    }
};

    auto do_perform = [&](bool insecure) -> CURLcode {
        response.clear();
        setup_ssl(insecure);
        return curl_easy_perform(curl);
    };

    CURLcode res = do_perform(false);
    long http_code = 0;
    curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &http_code);

    if (res != CURLE_OK) {
        bool ssl_fail = false;
#ifdef CURLE_PEER_FAILED_VERIFICATION
        ssl_fail = ssl_fail || (res == CURLE_PEER_FAILED_VERIFICATION);
#endif
#ifdef CURLE_SSL_CACERT
        ssl_fail = ssl_fail || (res == CURLE_SSL_CACERT);
#endif
#ifdef CURLE_SSL_CACERT_BADFILE
        ssl_fail = ssl_fail || (res == CURLE_SSL_CACERT_BADFILE);
#endif
#ifdef CURLE_SSL_ISSUER_ERROR
        ssl_fail = ssl_fail || (res == CURLE_SSL_ISSUER_ERROR);
#endif
#ifdef CURLE_SSL_CONNECT_ERROR
        ssl_fail = ssl_fail || (res == CURLE_SSL_CONNECT_ERROR);
#endif
        if (ssl_fail) {
            res = do_perform(true);
            curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &http_code);
        }
    }

    if (headers) curl_slist_free_all(headers);
    curl_easy_cleanup(curl);

    if (res != CURLE_OK) {
        if (out_err) {
            std::string eb = errbuf[0] ? std::string(errbuf) : std::string("");
            *out_err = "NETWORK: " + std::to_string((int)res) + " " + std::string(curl_easy_strerror(res));
            if (!eb.empty()) *out_err += " | " + eb;
        }
        return "";
    }

    if (response.empty()) {
        if (out_err) *out_err = (http_code ? ("EMPTY_RESPONSE (HTTP " + std::to_string(http_code) + ")") : "EMPTY_RESPONSE");
        return "";
    }

    if (http_code < 200 || http_code >= 300) {
        if (out_err) *out_err = "HTTP_" + std::to_string(http_code);
    }

    return response;
}

void try_login() {
    std::string key;
    {
        std::lock_guard<std::mutex> lock(login_lock);
        key = std::string(key_input);
        last_msg = "Checking key...";
    }

    key.erase(std::remove_if(key.begin(), key.end(), [](unsigned char c) { return std::isspace(c) != 0; }), key.end());
    for (char &c : key) c = (char)std::toupper((unsigned char)c);

    if (key.empty()) {
        std::lock_guard<std::mutex> lock(login_lock);
        login_ok = false;
        login_in_progress = false;
        last_msg = "Key is empty";
        Sunny_DisableFeaturesOnLogout();
        return;
    }

    if (!Sunny_IsValidKeyFormatSafe(key)) {
        std::lock_guard<std::mutex> lock(login_lock);
        login_ok = false;
        login_in_progress = false;
        last_msg = "Invalid key format (ex: SUNNY-ABCD-EFGH-IKNM)";
        Sunny_DisableFeaturesOnLogout();
        return;
    }

    std::string device = getDeviceID();
    if (device.empty()) device = "UNKNOWN_DEVICE";
    std::string device_name = GetDeviceNameSafe();

    std::string err;
    std::string response = send_request(key, device, device_name, &err);
  
    bool ok = false;
    if (response.empty()) {
        ok = false;
        if (err.empty()) err = "Network error";
    } else {
        ok = ParseVerifyResponseToState(response, &err);
    }

    {
        std::lock_guard<std::mutex> lock(login_lock);

        login_ok = ok;

        if (ok) {
            g_license_started   = gLicState.started;
            g_license_expires_at = gLicState.expires_at;
            g_license_server_time   = gLicState.server_time;

            if (gLicState.has_remaining) {
                g_remaining_seconds_at_sync = gLicState.remaining_seconds;            } else {
                g_remaining_seconds_at_sync = -1;            }

            if (LicenseIsExpiredOrInvalid()) {
                login_ok = false;
                last_msg = "Expired key. Please renew.";
                Sunny_DisableFeaturesOnLogout();
            } else {
                int remNow = GetRemainingSecondsNow();
                std::string remText = (remNow >= 0) ? FormatRemainingDHMS(remNow) : std::string("—");
                last_msg = "Login OK • Remaining: " + remText;
               
            }
        } else {
            last_msg = err.empty() ? "Login failed" : err;
            Sunny_DisableFeaturesOnLogout();
        }

        login_in_progress = false;
    }
}

static inline void JniClearException(JNIEnv* env) {
    if (env && env->ExceptionCheck()) env->ExceptionClear();
}

static jobject GetContextSafe(JNIEnv* env) {
    if (!env) return nullptr;

    if (g_Activity) return g_Activity;

    jclass unityPlayer = env->FindClass("com/unity3d/player/UnityPlayer");
    JniClearException(env);
    if (unityPlayer) {
        jfieldID f = env->GetStaticFieldID(unityPlayer, "currentActivity", "Landroid/app/Activity;");
        JniClearException(env);
        if (f) {
            jobject act = env->GetStaticObjectField(unityPlayer, f);
            JniClearException(env);
            if (act) {
                g_Activity = env->NewGlobalRef(act);
                return g_Activity;
            }
        }
    }

    jclass activityThreadClass = env->FindClass("android/app/ActivityThread");
    JniClearException(env);
    if (activityThreadClass) {
        jmethodID currentActivityThread = env->GetStaticMethodID(activityThreadClass, "currentActivityThread", "()Landroid/app/ActivityThread;");
        JniClearException(env);
        jobject at = currentActivityThread ? env->CallStaticObjectMethod(activityThreadClass, currentActivityThread) : nullptr;
        JniClearException(env);
        if (at) {
            jmethodID getApplication = env->GetMethodID(activityThreadClass, "getApplication", "()Landroid/app/Application;");
            JniClearException(env);
            jobject app = getApplication ? env->CallObjectMethod(at, getApplication) : nullptr;
            JniClearException(env);
            if (app) {
                g_Activity = env->NewGlobalRef(app);
                return g_Activity;
            }
        }
    }

    return nullptr;
}

static std::string GetClipboardSafe() {
    if (!jvm1) return "";

    JNIEnv* env = nullptr;
    bool needsDetach = false;

    jint jniStat = jvm1->GetEnv((void**)&env, JNI_VERSION_1_6); 
    if (jniStat == JNI_EDETACHED) {
        if (jvm1->AttachCurrentThread(&env, nullptr) != JNI_OK) return "";
        needsDetach = true;
    }
    if (!env) return "";

    std::string out;

    jobject ctx = GetContextSafe(env);
    if (!ctx) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jclass ctxCls = env->GetObjectClass(ctx);
    jmethodID midGetService = env->GetMethodID(ctxCls, "getSystemService", "(Ljava/lang/String;)Ljava/lang/Object;");
    JniClearException(env);
    if (!midGetService) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jstring svc = env->NewStringUTF("clipboard");
    jobject cm = env->CallObjectMethod(ctx, midGetService, svc);
    env->DeleteLocalRef(svc);
    JniClearException(env);

    if (!cm) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return "";
    }

    jclass cmCls = env->GetObjectClass(cm);

    jmethodID midHasPrimary = env->GetMethodID(cmCls, "hasPrimaryClip", "()Z");
    jmethodID midGetPrimary = env->GetMethodID(cmCls, "getPrimaryClip", "()Landroid/content/ClipData;");
    JniClearException(env);

    if (midHasPrimary && midGetPrimary && env->CallBooleanMethod(cm, midHasPrimary)) {
        jobject clipData = env->CallObjectMethod(cm, midGetPrimary);
        JniClearException(env);

        if (clipData) {
            jclass clipDataCls = env->GetObjectClass(clipData);
            jmethodID midGetItemAt = env->GetMethodID(clipDataCls, "getItemAt", "(I)Landroid/content/ClipData$Item;");
            JniClearException(env);

            if (midGetItemAt) {
                jobject item = env->CallObjectMethod(clipData, midGetItemAt, 0);
                JniClearException(env);

                if (item) {
                    jclass itemCls = env->GetObjectClass(item);

                    jmethodID midCoerce = env->GetMethodID(itemCls, "coerceToText", "(Landroid/content/Context;)Ljava/lang/CharSequence;");
                    jmethodID midGetText = env->GetMethodID(itemCls, "getText", "()Ljava/lang/CharSequence;");
                    JniClearException(env);

                    jobject textObj = nullptr;
                    if (midCoerce) textObj = env->CallObjectMethod(item, midCoerce, ctx);
                    JniClearException(env);
                    if (!textObj && midGetText) textObj = env->CallObjectMethod(item, midGetText);
                    JniClearException(env);

                    if (textObj) {
                        jclass csCls = env->GetObjectClass(textObj);
                        jmethodID midToString = env->GetMethodID(csCls, "toString", "()Ljava/lang/String;");
                        JniClearException(env);
                        if (midToString) {
                            jstring jstr = (jstring)env->CallObjectMethod(textObj, midToString);
                            JniClearException(env);
                            if (jstr) {
                                const char* c = env->GetStringUTFChars(jstr, nullptr);
                                if (c) {
                                    out = c;
                                    env->ReleaseStringUTFChars(jstr, c);
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    if (out.empty()) {
        jmethodID midGetTextOld = env->GetMethodID(cmCls, "getText", "()Ljava/lang/CharSequence;");
        JniClearException(env);
        if (midGetTextOld) {
            jobject textObj = env->CallObjectMethod(cm, midGetTextOld);
            JniClearException(env);
            if (textObj) {
                jclass csCls = env->GetObjectClass(textObj);
                jmethodID midToString = env->GetMethodID(csCls, "toString", "()Ljava/lang/String;");
                JniClearException(env);
                if (midToString) {
                    jstring jstr = (jstring)env->CallObjectMethod(textObj, midToString);
                    JniClearException(env);
                    if (jstr) {
                        const char* c = env->GetStringUTFChars(jstr, nullptr);
                        if (c) {
                            out = c;
                            env->ReleaseStringUTFChars(jstr, c);
                        }
                    }
                }
            }
        }
    }

    if (needsDetach) jvm1->DetachCurrentThread();
    return out;
}

static void TrimLineInPlace(std::string& s) {
    while (!s.empty() && (s.back() == '\n' || s.back() == '\r' || s.back() == ' ' || s.back() == '\t')) s.pop_back();
    size_t i = 0;
    while (i < s.size() && (s[i] == ' ' || s[i] == '\t')) i++;
    if (i > 0) s.erase(0, i);
}

static void LoadLoginUiConfigOnce() {
    if (g_loginui_inited) return;
    g_loginui_inited = true;

    std::string path = GetLoginUiCfgPath();
    if (path.empty()) return;

    FILE* f = fopen(path.c_str(), "rb");
    if (!f) return;

    char line[512];
    while (fgets(line, sizeof(line), f)) {
        std::string s(line);
        TrimLineInPlace(s);
        if (s.rfind("auto=", 0) == 0) {
            g_loginui_auto_key = (s.size() >= 6 && s[5] == '1');
        } else if (s.rfind("save=", 0) == 0) {
            g_loginui_save_key = (s.size() >= 6 && s[5] == '1');
        } else if (s.rfind("key=", 0) == 0) {
            std::string k = s.substr(4);
            TrimLineInPlace(k);
            if (!k.empty()) {
                snprintf(key_input, sizeof(key_input), "%s", k.c_str());
            g_loginui_saved_key = k;
            }
        }
    }
    if (!g_loginui_save_key) {
        key_input[0] = '\0';
        g_loginui_saved_key.clear();
    }

    fclose(f);
}

static inline void OpenUrlSafe_JNI(const char* url) { 
    if (!url || !*url) return;
    if (!jvm1) return;

    JNIEnv* env = nullptr;
    bool needsDetach = false;

    jint jniStat = jvm1->GetEnv((void**)&env, JNI_VERSION_1_6); 
    if (jniStat == JNI_EDETACHED) {
        if (jvm1->AttachCurrentThread(&env, nullptr) != JNI_OK) return;
        needsDetach = true;
    }
    if (!env) return;

    jobject ctx = GetContextSafe(env);
    if (!ctx) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return;
    }

    jclass intentCls = env->FindClass("android/content/Intent");
    jclass uriCls = env->FindClass("android/net/Uri");
    JniClearException(env);

    if (!intentCls || !uriCls) {
        if (needsDetach) jvm1->DetachCurrentThread();
        return;
    }

    jfieldID fidActionView = env->GetStaticFieldID(intentCls, "ACTION_VIEW", "Ljava/lang/String;");
    JniClearException(env);
    jstring actionView = fidActionView ? (jstring)env->GetStaticObjectField(intentCls, fidActionView) : nullptr;
    JniClearException(env);

    jmethodID midParse = env->GetStaticMethodID(uriCls, "parse", "(Ljava/lang/String;)Landroid/net/Uri;");
    JniClearException(env);

    jstring jurl = env->NewStringUTF(url);
    jobject uriObj = (midParse && jurl) ? env->CallStaticObjectMethod(uriCls, midParse, jurl) : nullptr;
    JniClearException(env);

    jmethodID ctor = env->GetMethodID(intentCls, "<init>", "(Ljava/lang/String;Landroid/net/Uri;)V");
    JniClearException(env);
    jobject intentObj = (ctor && actionView && uriObj) ? env->NewObject(intentCls, ctor, actionView, uriObj) : nullptr;
    JniClearException(env);

    if (intentObj) {
        // Ensure NEW_TASK when ctx is Application (fallback)
        jfieldID fidNewTask = env->GetStaticFieldID(intentCls, "FLAG_ACTIVITY_NEW_TASK", "I");
        JniClearException(env);
        jint newTask = fidNewTask ? env->GetStaticIntField(intentCls, fidNewTask) : 0;
        JniClearException(env);
        jmethodID midAddFlags = env->GetMethodID(intentCls, "addFlags", "(I)Landroid/content/Intent;");
        JniClearException(env);
        if (midAddFlags && newTask) {
            env->CallObjectMethod(intentObj, midAddFlags, newTask);
            JniClearException(env);
        }

        jclass actCls = env->GetObjectClass(ctx);
        jmethodID midStart = env->GetMethodID(actCls, "startActivity", "(Landroid/content/Intent;)V");
        JniClearException(env);
        if (midStart) {
            env->CallVoidMethod(ctx, midStart, intentObj);
            JniClearException(env);
        }
    }

    if (jurl) env->DeleteLocalRef(jurl);
    if (needsDetach) jvm1->DetachCurrentThread();
}

static ImU32 RgbHSV(float h, float s, float v, float a) {
    ImColor c = ImColor::HSV(h, s, v, a);
    return c;
}

static ImVec2 PerimeterPointRect(ImVec2 p0, ImVec2 p1, float d) {
    float w = p1.x - p0.x;
    float h = p1.y - p0.y;
    float L = 2.0f * (w + h);
    if (L <= 0.0f) return p0;
    while (d < 0.0f) d += L;
    while (d > L) d -= L;

    if (d <= w) {
        return ImVec2(p0.x + d, p0.y);
    }
    d -= w;
    if (d <= h) {
        return ImVec2(p1.x, p0.y + d);
    }
    d -= h;
    if (d <= w) {
        return ImVec2(p1.x - d, p1.y);
    }
    d -= w;
    return ImVec2(p0.x, p1.y - d);
}

static void DrawRainbowWindowDecor(ImVec2 wp, ImVec2 ws, float rounding) {
    ImDrawList* dl = ImGui::GetWindowDrawList();
    if (!dl) return;

    const float t = (float)ImGui::GetTime();
    const float alphaBg = 0.90f;     
    const float alphaGlow = 0.22f;   
    const float alphaBorder = 0.95f; 

    ImVec2 p0(wp.x + 6.0f, wp.y + 6.0f);
    ImVec2 p1(wp.x + ws.x - 6.0f, wp.y + ws.y - 6.0f);

    dl->AddRectFilled(p0, p1, IM_COL32(18, 18, 22, (int)(255.0f * alphaBg)), rounding);

    dl->AddRect(p0, p1, IM_COL32(255, 255, 255, (int)(255.0f * alphaGlow)), rounding, 0, 2.0f);

    float w = (p1.x - p0.x);
    float h = (p1.y - p0.y);
    float L = 2.0f * (w + h);
    if (L < 10.0f) return;

    const int seg = 220;
    const float speed = 0.18f; 
    auto posOnRect = [&](float u) -> ImVec2 {
        float d = u * L;
        if (d < w) return ImVec2(p0.x + d, p0.y);
        d -= w;
        if (d < h) return ImVec2(p1.x, p0.y + d);
        d -= h;
        if (d < w) return ImVec2(p1.x - d, p1.y);
        d -= w;
        return ImVec2(p0.x, p1.y - d);
    };

    ImVec2 prev = posOnRect(0.0f);
    for (int i = 1; i <= seg; i++) {
        float u0 = (float)(i - 1) / (float)seg;
        float u1 = (float)i / (float)seg;

        float hue = fmodf(t * speed + u0, 1.0f);
        ImU32 col = RgbHSV(hue, 1.0f, 1.0f, alphaBorder);

        ImVec2 cur = posOnRect(u1);
        dl->AddLine(prev, cur, col, 3.0f);
        prev = cur;
    }

    dl->AddRect(ImVec2(p0.x + 2.0f, p0.y + 2.0f), ImVec2(p1.x - 2.0f, p1.y - 2.0f),
                IM_COL32(0, 0, 0, 140), rounding - 1.0f, 0, 1.0f);
}

static void DrawEyeIcon(ImDrawList* dl, ImVec2 pmin, ImVec2 pmax, ImU32 col, bool isOpen) {
    if (!dl) return;
    float w = pmax.x - pmin.x;
    float h = pmax.y - pmin.y;
    ImVec2 center(pmin.x + w * 0.5f, pmin.y + h * 0.5f);
    
    float eyeW = w * 0.65f;
    float eyeH = h * 0.45f;
    
    ImVec2 left(center.x - eyeW * 0.5f, center.y);
    ImVec2 right(center.x + eyeW * 0.5f, center.y);

    const int segments = 12; 
    for (int i = 0; i < segments; i++) {
        float t1 = (float)i / segments;
        float t2 = (float)(i + 1) / segments;
        
        auto calcQ = [](ImVec2 p0, ImVec2 p1, ImVec2 p2, float t) {
            float u = 1.0f - t;
            return ImVec2(u*u*p0.x + 2*u*t*p1.x + t*t*p2.x, u*u*p0.y + 2*u*t*p1.y + t*t*p2.y);
        };
        ImVec2 pA1 = calcQ(left, ImVec2(center.x, center.y - eyeH), right, t1);
        ImVec2 pA2 = calcQ(left, ImVec2(center.x, center.y - eyeH), right, t2);
        dl->AddLine(pA1, pA2, col, 1.8f);
        ImVec2 pB1 = calcQ(left, ImVec2(center.x, center.y + eyeH), right, t1);
        ImVec2 pB2 = calcQ(left, ImVec2(center.x, center.y + eyeH), right, t2);
        dl->AddLine(pB1, pB2, col, 1.8f);
    }
    
    if (isOpen) {
        dl->AddCircle(center, eyeH * 0.45f, col, 12, 1.8f);
        dl->AddCircleFilled(center, eyeH * 0.20f, col, 12);
    } else {
        dl->AddLine(ImVec2(center.x - eyeW * 0.45f, center.y - eyeH * 0.8f), 
                    ImVec2(center.x + eyeW * 0.45f, center.y + eyeH * 0.8f), 
                    col, 2.0f);
    }
}

// ===== HSV -> RGB =====
inline ImVec4 HSVtoRGB(float h, float s, float v, float a = 1.0f)
{
    float r, g, b;
    ImGui::ColorConvertHSVtoRGB(h, s, v, r, g, b);
    return ImVec4(r, g, b, a);
}

// ===== NUTNHANH / TOGGLE SWITCH (SIZE TO HƠN) =====
inline bool Nutnhanh(const char* id, bool* v, ImVec2 switch_size = ImVec2(45, 22))
{
    ImGuiWindow* window = ImGui::GetCurrentWindow();
    if (window->SkipItems)
        return false;

    ImGuiContext& g = *ImGui::GetCurrentContext();
    ImGuiID toggle_id = window->GetID(id);

    // ===== SCALE SWITCH (TO LÊN) =====
    float scale = 1.3f;              // 👈 chỉnh 1.2 ~ 1.5 tùy gu
    switch_size.x *= scale;
    switch_size.y *= scale;

    // ===== CONFIG =====
    ImVec2 frame_padding(12.0f, 12.0f);
    float text_spacing = 6.0f;

    ImVec2 text_size = ImGui::CalcTextSize(id);

    // ===== FRAME SIZE =====
    float frame_w = (switch_size.x > text_size.x ? switch_size.x : text_size.x)
                    + frame_padding.x * 2.0f;
    float frame_h = switch_size.y + text_spacing + text_size.y
                    + frame_padding.y * 2.0f;

    ImVec2 pos = window->DC.CursorPos;

    ImVec2 frame_max(
        pos.x + frame_w,
        pos.y + frame_h
    );

    ImRect outer_bb(pos, frame_max);

    ImGui::ItemSize(outer_bb);
    if (!ImGui::ItemAdd(outer_bb, toggle_id))
        return false;

    // ===== SWITCH RECT (CENTER) =====
    ImVec2 switch_pos(
        outer_bb.Min.x + (frame_w - switch_size.x) * 0.5f,
        outer_bb.Min.y + frame_padding.y
    );

    ImVec2 switch_max(
        switch_pos.x + switch_size.x,
        switch_pos.y + switch_size.y
    );

    ImRect bb(switch_pos, switch_max);

    bool hovered, held;
    bool pressed = ImGui::ButtonBehavior(bb, toggle_id, &hovered, &held);
    if (pressed)
        *v = !(*v);

    // ===== ANIMATION =====
    ImGuiStorage* storage = ImGui::GetStateStorage();
    float anim = storage->GetFloat(toggle_id, *v ? 1.0f : 0.0f);
    float target = *v ? 1.0f : 0.0f;

    anim = ImLerp(anim, target, g.IO.DeltaTime * 12.0f);
    storage->SetFloat(toggle_id, anim);

    ImDrawList* draw = ImGui::GetWindowDrawList();

    // ===== COLORS =====
    ImVec4 col_bg_off = ImVec4(0.18f, 0.18f, 0.20f, 1.0f);
    ImVec4 col_bg_on  = ImVec4(0.0f, 0.48f, 1.0f, 1.0f);
    ImVec4 col_knob   = ImVec4(1, 1, 1, 1);

    float radius = switch_size.y * 0.5f;

    // ===== SWITCH BACKGROUND =====
    ImVec4 bg = ImLerp(col_bg_off, col_bg_on, anim);
    draw->AddRectFilled(bb.Min, bb.Max, ImGui::GetColorU32(bg), radius);

    // ===== KNOB =====
    float pad = 2.5f;
    float knob_r = radius - pad;

    float knob_x = ImLerp(
        bb.Min.x + pad + knob_r,
        bb.Max.x - pad - knob_r,
        anim
    );

    ImVec2 knob_center(
        knob_x,
        bb.Min.y + radius
    );

    // Shadow
    draw->AddCircleFilled(
        ImVec2(knob_center.x, knob_center.y + 1.8f),
        knob_r,
        IM_COL32(0, 0, 0, 90)
    );

    // Knob
    draw->AddCircleFilled(
        knob_center,
        knob_r,
        ImGui::GetColorU32(col_knob)
    );

    // Hover glow
    if (hovered)
        draw->AddCircleFilled(
            knob_center,
            knob_r,
            IM_COL32(255, 255, 255, 35)
        );

    // ===== TEXT BELOW SWITCH =====
    ImVec2 text_pos(
        outer_bb.Min.x + (frame_w - text_size.x) * 0.5f,
        bb.Max.y + text_spacing
    );

    draw->AddText(text_pos, IM_COL32(210, 210, 210, 255), id);

    // ===== 🌈 RAINBOW BORDER (ONLY BORDER) =====
    static float hue = 0.0f;
    hue += g.IO.DeltaTime * 0.25f;
    if (hue > 1.0f) hue -= 1.0f;

    ImVec4 rainbow = HSVtoRGB(hue, 1.0f, 1.0f);

    draw->AddRect(
        outer_bb.Min,
        outer_bb.Max,
        ImGui::GetColorU32(rainbow),
        12.0f,   // bo góc
        0,
        2.4f     // độ dày viền
    );

    return pressed;
}


void DrawSpinner(ImVec2 center, float radius, int thickness, const ImVec4& color, float speed) {
    ImGuiWindow* window = ImGui::GetCurrentWindow();
    if (window->SkipItems) return;
    
    ImDrawList* drawList = window->DrawList;
    
    // Animation dựa trên thời gian thực
    const float time = (float)ImGui::GetTime() * speed;
    const int num_segments = 30;
    const float start_angle = time;
    const float end_angle = start_angle + (IM_PI * 1.5f); // Vòng cung 270 độ

    // Vẽ vòng mờ background (để tạo độ sâu)
    drawList->AddCircle(center, radius, ImGui::GetColorU32(ImVec4(color.x, color.y, color.z, 0.1f)), num_segments, (float)thickness);

    drawList->PathClear();
    drawList->PathArcTo(center, radius, start_angle, end_angle, num_segments);
    drawList->PathStroke(ImGui::GetColorU32(color), 0, (float)thickness);
}

void ToggleSwitch(const char* str_id, bool* v)
{
    ImVec2 p = ImGui::GetCursorScreenPos();
    ImDrawList* draw_list = ImGui::GetWindowDrawList();

    float height = 30.0f; // Set a custom height (reduce the size here)
    float width = height * 2.0f; // Width remains proportional to the height
    float radius = height * 0.50f;

    ImGui::InvisibleButton(str_id, ImVec2(width, height));
    if (ImGui::IsItemClicked())
        *v = !*v;

    float t = *v ? 1.0f : 0.0f;

    ImGuiContext& g = *GImGui;
    float ANIM_SPEED = 0.08f;
    if (g.LastActiveId == g.CurrentWindow->GetID(str_id))
    {
        float t_anim = ImSaturate(g.LastActiveIdTimer / ANIM_SPEED);
        t = *v ? (t_anim) : (1.0f - t_anim);
    }

    ImU32 col_bg = *v
        ? ImGui::GetColorU32(ImVec4(0.26f, 0.59f, 0.98f, 0.80f)) // Green when ON
        : ImGui::GetColorU32(ImVec4(0.5f, 0.5f, 0.5f, 1.00f));   // Real Red when OFF

    draw_list->AddRectFilled(p, ImVec2(p.x + width, p.y + height), col_bg, height * 0.5f);
    draw_list->AddCircleFilled(ImVec2(p.x + radius + t * (width - radius * 2.0f), p.y + radius), radius - 5.0f, IM_COL32(255, 255, 255, 255));

    // Display the toggle label without ON/OFF text
    ImGui::SameLine();
    ImGui::Text("%s", str_id);
}
 

#include <cmath>

// --- Helper Math ---
ImVec4 LerpColor(ImVec4 a, ImVec4 b, float t) {
    return ImVec4(a.x + (b.x - a.x) * t, a.y + (b.y - a.y) * t, a.z + (b.z - a.z) * t, a.w + (b.w - a.w) * t);
}

// Hàm vẽ Tab với đôi tai mèo ở phía trên
bool DrawTag(const char* label, bool selected, float width, float height = 35.0f, ImColor themeColor = ImColor(255, 150, 200)) // Mặc định màu hồng cho hợp tai mèo
{
    ImGuiWindow* window = ImGui::GetCurrentWindow();
    if (window->SkipItems) return false;

    ImGuiContext& g = *GImGui;
    const ImGuiID id = window->GetID(label);
    const ImVec2 label_size = ImGui::CalcTextSize(label, NULL, true);

    ImVec2 pos = window->DC.CursorPos;
    ImRect bb(pos, ImVec2(pos.x + width, pos.y + height));
    
    // Xử lý Input
    bool clicked = ImGui::InvisibleButton(label, ImVec2(width, height));
    bool hovered = ImGui::IsItemHovered();

    ImDrawList* draw = window->DrawList;

    // --- Animation Logic ---
    static std::unordered_map<ImGuiID, float> select_anim;
    float dt = ImGui::GetIO().DeltaTime;
    
    if (selected) select_anim[id] = std::min(1.0f, select_anim[id] + dt * 6.0f);
    else select_anim[id] = std::max(0.0f, select_anim[id] - dt * 5.0f);
    
    float t = select_anim[id];
    ImVec4 accent = themeColor.Value;

    // --- 1. VẼ TAI MÈO (Vẽ trước để nằm dưới viền hoặc trên tùy ý, ở đây vẽ trên) ---
    if (t > 0.01f) 
    {
        float earSize = 12.0f * t; // Tai to dần khi được chọn
        float earOffset = 8.0f;    // Khoảng cách từ mép nút vào trong
        ImU32 earCol = ImGui::GetColorU32(ImVec4(accent.x, accent.y, accent.z, t));
        ImU32 earInnerCol = ImGui::GetColorU32(ImVec4(1.0f, 1.0f, 1.0f, 0.4f * t)); // Màu bên trong tai

        // Tai trái
        ImVec2 L1(bb.Min.x + earOffset, bb.Min.y + 2.0f);
        ImVec2 L2(bb.Min.x + earOffset + earSize * 0.5f, bb.Min.y - earSize);
        ImVec2 L3(bb.Min.x + earOffset + earSize, bb.Min.y + 2.0f);
        draw->AddTriangleFilled(L1, L2, L3, earCol);
        // Vẽ thêm một tam giác nhỏ hơn bên trong cho điệu
        draw->AddTriangleFilled(L1, ImVec2(L2.x, L2.y + 4.0f), L3, earInnerCol);

        // Tai phải
        ImVec2 R1(bb.Max.x - earOffset - earSize, bb.Min.y + 2.0f);
        ImVec2 R2(bb.Max.x - earOffset - earSize * 0.5f, bb.Min.y - earSize);
        ImVec2 R3(bb.Max.x - earOffset, bb.Min.y + 2.0f);
        draw->AddTriangleFilled(R1, R2, R3, earCol);
        draw->AddTriangleFilled(R1, ImVec2(R2.x, R2.y + 4.0f), R3, earInnerCol);
    }

    // --- 2. VẼ NỀN NÚT ---
    float rounding = 8.0f;
    draw->AddRectFilled(bb.Min, bb.Max, ImGui::GetColorU32(ImVec4(0.12f, 0.12f, 0.14f, 1.0f)), rounding);
    
    if (t > 0.01f) {
        draw->AddRectFilled(bb.Min, bb.Max, ImGui::GetColorU32(ImVec4(accent.x, accent.y, accent.z, 0.15f * t)), rounding);
    }

    // --- 3. HIỆU ỨNG GẠCH CHÉO / (AUTO SCAN) ---
    if (selected) {
        draw->PushClipRect(bb.Min, bb.Max, true);
        float sweepTime = fmodf((float)ImGui::GetTime(), 2.5f);
        if (sweepTime < 1.5f) {
            float st = sweepTime / 1.5f;
            float slant = 20.0f;
            float thick = 10.0f;
            float startX = bb.Min.x - thick - slant;
            float endX = bb.Max.x + thick + slant;
            float cx = startX + (endX - startX) * st;

            draw->AddQuadFilled(
                ImVec2(cx - thick + slant, bb.Min.y),
                ImVec2(cx + thick + slant, bb.Min.y),
                ImVec2(cx + thick - slant, bb.Max.y),
                ImVec2(cx - thick - slant, bb.Max.y),
                ImGui::GetColorU32(ImVec4(1,1,1, 0.3f))
            );
        }
        draw->PopClipRect();
    }

    // --- 4. VIỀN VÀ CHỮ ---
    ImU32 borderCol = ImGui::GetColorU32(LerpColor(ImVec4(0.3f, 0.3f, 0.3f, 0.5f), accent, t));
    draw->AddRect(bb.Min, bb.Max, borderCol, rounding, 0, 1.5f);

    ImVec2 textPos(bb.Min.x + (width - label_size.x) * 0.5f, bb.Min.y + (height - label_size.y) * 0.5f);
    draw->AddText(textPos, ImGui::GetColorU32(LerpColor(ImVec4(0.6f, 0.6f, 0.6f, 1.0f), ImVec4(1,1,1,1), t)), label);

    return clicked;
}




static void DrawHeaderLine(const char* label,
                                float short_len = 15.0f,
                                float spacing   = 4.0f,
                                float thickness = 1.0f,
                                float fontScale = 0.8f)   // thu nhỏ chữ
{
    ImGuiWindow* window = ImGui::GetCurrentWindow();
    if (!window) return;
    ImDrawList* draw = ImGui::GetWindowDrawList();

    ImGui::SetWindowFontScale(fontScale);

    ImVec2 startPos = ImGui::GetCursorScreenPos();
    ImVec2 textSize = ImGui::CalcTextSize(label);
    float line_y = startPos.y + textSize.y * 0.5f;

    // line trái (màu trắng)
    draw->AddLine(
        ImVec2(startPos.x, line_y),
        ImVec2(startPos.x + short_len, line_y),
        IM_COL32(255,255,255,255), thickness
    );

    // text (màu trắng)
    ImVec2 textPos = { startPos.x + short_len + 4.0f, startPos.y };
    draw->AddText(textPos, IM_COL32(255,255,255,255), label);

    // line phải (màu trắng)
    ImVec2 right_a = { textPos.x + textSize.x + spacing, line_y };
    ImVec2 right_b = { startPos.x + ImGui::GetContentRegionAvail().x, line_y };
    if (right_b.x > right_a.x)
        draw->AddLine(right_a, right_b, IM_COL32(255,255,255,255), thickness);

    ImGui::SetWindowFontScale(1.0f);
    ImGui::Dummy(ImVec2(ImGui::GetContentRegionAvail().x, textSize.y + 2.0f));
}

    bool Switchx(const char* label, bool* v, float inner_padding = 3.0f, float scale = 1.4f)
    {
        ImGuiWindow* window = GetCurrentWindow();
        if (window->SkipItems)
            return false;

        ImGuiContext& g = *GImGui;
        const ImGuiStyle& style = g.Style;
        const ImGuiID id = window->GetID(label);

        // 1. CẤU HÌNH KÍCH THƯỚC
        // ---------------------------------------------------------
        // Kích thước Switch
        float base_height = GetFrameHeight();
        float switch_height = base_height * scale;
        float switch_width  = switch_height * 2.0f;

        // Kích thước Text Box (bên dưới)
        ImVec2 text_size = CalcTextSize(label, NULL, true);
        float text_padding_y = 4.0f; // Padding trên/dưới của text
        float gap_height = 4.0f;     // Khoảng cách giữa Switch và Text Box
        float text_box_height = text_size.y + text_padding_y * 2.0f;

        // Tổng kích thước widget (Switch + Gap + Text Box)
        // Chiều rộng lấy theo switch (hoặc text nếu text dài hơn, ở đây ta cố định theo switch cho đẹp)

        // Nếu text quá dài so với switch, bạn có thể uncomment dòng dưới để auto-resize width
        // float total_w = ImMax(switch_width, text_size.x + 10.0f); 

        float total_h = switch_height + gap_height + text_box_height;

        ImVec2 pos = window->DC.CursorPos;
        ImVec2 size(total_w, total_h);
        const ImRect total_bb(pos, ImVec2(pos.x + size.x, pos.y + size.y));

        // QUAN TRỌNG: Khai báo kích thước tổng cho ImGui biết để không bị đè item sau
        ItemSize(total_bb, style.FramePadding.y);
        if (!ItemAdd(total_bb, id))
            return false;

        // 2. XỬ LÝ INPUT
        // ---------------------------------------------------------

        bool pressed = ButtonBehavior(total_bb, id, &hovered, &held);
        if (pressed)
        {
            *v = !(*v);
            MarkItemEdited(id);
        }

        // Animation (dựa trên trạng thái hover/active)
        float t = *v ? 1.0f : 0.0f;
        // Nếu muốn animation mượt (cần ImGui state storage), hiện tại dùng t = 0 hoặc 1
        
        // 3. VẼ NÚT SWITCH (PHẦN TRÊN)

        float inner_diameter = switch_height - inner_padding * 2.0f;
        float radius = inner_diameter * 0.5f;
        float move_range = (switch_width - inner_padding * 2.0f) - inner_diameter;

        );

            text_bg_bb.Min.y + (text_bg_bb.GetHeight() - text_size.y) * 0.5f
        );
        
        RenderText(text_pos, label);

        return pressed;
    }

// Hàm ToggleSwitch Siêu Đẹp
inline bool ToggleSwitch1(const char* id, bool* v, ImVec2 size_arg = ImVec2(45, 22))
{
    ImGuiWindow* window = ImGui::GetCurrentWindow();
    if (window->SkipItems)
        return false;

    ImGuiContext& g = *ImGui::GetCurrentContext();
    const ImGuiStyle& style = g.Style;
    ImGuiID toggle_id = window->GetID(id);
    
    // Tính toán kích thước (mặc định nếu size truyền vào là 0)
    ImVec2 size = size_arg;
    if (size.x == 0) size.x = 45;
    if (size.y == 0) size.y = 22;

    ImVec2 pos = window->DC.CursorPos;
    ImRect bb(pos, ImVec2(pos.x + size.x, pos.y + size.y));
    
    ImGui::ItemSize(bb, style.FramePadding.y);
    if (!ImGui::ItemAdd(bb, toggle_id))
        return false;

    bool hovered, held;
    bool pressed = ImGui::ButtonBehavior(bb, toggle_id, &hovered, &held);
    if (pressed)
        *v = !(*v);

    // --- ANIMATION LOGIC ---
    // Sử dụng StateStorage để lưu trạng thái animation hiện tại
    ImGuiStorage* storage = ImGui::GetStateStorage();
    float anim = storage->GetFloat(toggle_id, *v ? 1.0f : 0.0f);
    float target = *v ? 1.0f : 0.0f;

    // Tốc độ animation (càng cao càng nhanh)
    float speed = 15.0f * g.IO.DeltaTime; 
    
    // Xử lý animation mượt (Ease-out effect đơn giản)
    if (anim != target) {
        float dir = (target > anim) ? 1.0f : -1.0f;
        anim += speed * dir;
        // Kẹp giá trị trong khoảng [0, 1] và đúng target
        if ((dir > 0 && anim > target) || (dir < 0 && anim < target)) 
            anim = target;
        
        storage->SetFloat(toggle_id, anim);
    }

    ImDrawList* draw = ImGui::GetWindowDrawList();

    // --- CẤU HÌNH MÀU SẮC & THÔNG SỐ ---
    
    // Màu nền khi TẮT (Xám đậm sang trọng)
    ImVec4 col_bg_off = ImVec4(0.18f, 0.18f, 0.2f, 1.0f); 
    // Màu nền khi BẬT (Xanh dương neon hoặc màu chủ đạo của bạn)
    ImVec4 col_bg_on  = ImVec4(0.0f, 0.48f, 1.0f, 1.0f);  // Màu xanh kiểu iOS
    // Màu của nút tròn (Knob)
    ImVec4 col_circle = ImVec4(1.0f, 1.0f, 1.0f, 1.0f);
    
    // Radius bo tròn
    float radius = size.y * 0.5f;

    // --- VẼ BACKGROUND (TRACK) ---
    
    // Tính toán màu nền hiện tại dựa trên animation
    ImVec4 bg_current = LerpColor(col_bg_off, col_bg_on, anim);
    
    // Hiệu ứng GLOW nhẹ khi bật (vẽ một hình chữ nhật mờ lớn hơn một chút)
    if (anim > 0.1f) {
        ImVec4 col_glow = col_bg_on;
        col_glow.w = 0.25f * anim; // Alpha giảm dần
        draw->AddRectFilled(
            ImVec2(bb.Min.x - 2, bb.Min.y - 2), 
            ImVec2(bb.Max.x + 2, bb.Max.y + 2), 
            ImGui::GetColorU32(col_glow), 
            radius + 2
        );
    }

    // Vẽ nền chính
    draw->AddRectFilled(bb.Min, bb.Max, ImGui::GetColorU32(bg_current), radius);
    
    // Thêm viền mỏng (Stroke) để tạo độ nét
    draw->AddRect(bb.Min, bb.Max, ImGui::GetColorU32(ImVec4(1,1,1,0.1f)), radius, 0, 1.0f);

    // --- VẼ NÚT TRÒN (KNOB) ---
    
    // Tính toán vị trí nút tròn
    // Padding để nút không chạm sát mép (trông tinh tế hơn)
    float pad = 2.0f; 
    float circle_radius = radius - pad;
    
    // Vị trí X chạy từ trái sang phải
    float knob_x_min = bb.Min.x + pad + circle_radius;
    float knob_x_max = bb.Max.x - pad - circle_radius;
    float knob_current_x = knob_x_min + (knob_x_max - knob_x_min) * anim;

    ImVec2 center(knob_current_x, bb.Min.y + radius);

    // BÓNG ĐỔ (Shadow) cho nút tròn
    // Vẽ một hình tròn mờ màu đen lệch xuống dưới một chút
    draw->AddCircleFilled(
        ImVec2(center.x, center.y + 1.5f), // Lệch Y xuống 1.5px
        circle_radius, 
        IM_COL32(0, 0, 0, 80) // Màu đen mờ (Alpha 80)
    );

    // Vẽ nút tròn chính
    draw->AddCircleFilled(center, circle_radius, ImGui::GetColorU32(col_circle));

    // (Tùy chọn) Hiệu ứng hover: Làm sáng nút tròn một chút khi di chuột vào
    if (hovered) {
        draw->AddCircleFilled(center, circle_radius, IM_COL32(255, 255, 255, 30));
    }

    return pressed;
}

struct RainBolt {
    ImVec2 start;       // Điểm bắt đầu (trên Logo)
    ImVec2 end;         // Điểm kết thúc (trên Text)
    float lifeTime;     // Thời gian tồn tại còn lại
    float maxLife;      // Tổng thời gian tồn tại (để tính độ mờ)
    float width;        // Độ dày tia sét
    ImU32 color;        // Màu sắc
    std::vector<float> offsets; // Lưu các điểm gãy khúc (jitter)
};

struct RainState {
    std::vector<RainBolt> bolts; // Danh sách các tia sét đang rơi
};

// Hàm random
float GetRand(float min, float max) {
    return min + static_cast <float> (rand()) / (static_cast <float> (RAND_MAX / (max - min)));
}

// Hàm vẽ 1 tia sét dựa trên dữ liệu đã lưu
void DrawRainBoltLine(ImDrawList* draw, const RainBolt& bolt) {
    int segments = 6; // Số khúc gãy
    ImVec2 current = bolt.start;
    float dx = bolt.end.x - bolt.start.x;
    float dy = bolt.end.y - bolt.start.y;

    // Tính độ trong suốt dựa trên thời gian sống (Fade out)
    float alphaRatio = bolt.lifeTime / bolt.maxLife;
    ImU32 fadedColor = (bolt.color & 0x00FFFFFF) | ((int)(255 * alphaRatio) << 24);

    for (int i = 1; i <= segments; i++) {
        float t = (float)i / segments;
        ImVec2 target = ImVec2(bolt.start.x + dx * t, bolt.start.y + dy * t);

        // Lấy offset nhiễu từ dữ liệu đã random trước đó (để tia sét không bị rung lắc quá mức mỗi frame)
        if (i < segments && i - 1 < bolt.offsets.size()) {
            target.x += bolt.offsets[i - 1];
        }

        draw->AddLine(current, target, fadedColor, bolt.width);
        current = target;
    }
    
    // Vẽ điểm nổ (Spark) ở cuối tia sét
    draw->AddCircleFilled(bolt.end, bolt.width * 1.5f, fadedColor);
}
